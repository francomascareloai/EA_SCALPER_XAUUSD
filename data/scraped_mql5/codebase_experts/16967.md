In this code the lot is not set rigidly - it is calculated each time in the function

//+------------------------------------------------------------------+  
//||  
//+------------------------------------------------------------------+  
double LotsOptimized()  
  {  
//--- select ExtLot size  
   ExtLot=NormalizeDouble(m\_account.FreeMargin()\*Risk/100,1);  
   ExtLot=LotCheck(ExtLot);  
//--- return ExtLot size  
   return(ExtLot);  
  }
