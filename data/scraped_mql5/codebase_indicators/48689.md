Example:

```mql5
#include <Comm.mqh>
int _x = 0, _y = 10, _z = 100;
//+------------------------------------------------------------------+
int OnInit(void)
  {
   EventSetMillisecondTimer(30);
   return(INIT_SUCCEEDED);
  }
//+------------------------------------------------------------------+
void OnTimer()
  {
   Comm(StringFormat("x=%d\ny=%d\nz=%d", _x, _y, _z),clrYellow,50);
   _x++;
   _y += 10;
   _z += 100;
  }
//+------------------------------------------------------------------+
void OnDeinit(const int reason)
  {
   EventKillTimer();
   DeinitComm();
  }
//+------------------------------------------------------------------+
```
